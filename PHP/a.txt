        function displayItems(data) {
            var products = document.querySelector('.products')
            for (var i = 0; i < data.length; i++) {
                var itemLength = data[i].items.length;
                if (itemLength) {
                    var heading = document.createElement('h2');
                    heading.textContent = `${data[i].Name}`;
                    products.appendChild(heading);
                    var itemsContainer = document.createElement('div');
                    itemsContainer.classList.add('items');
                    products.appendChild(itemsContainer);
                    for (var j = 0; j < data[i].items.length; j++) {
                        var card = document.createElement('div');
                        card.classList.add('card');
                        card.innerHTML = `<img src="./adminPages/assets/img/${data[i].items[j].ImagePath}"\>
                <h3>${data[i].items[j].Name}</h3>
                <span>Price:- ${data[i].items[j].Price}</span>
                <button>Add to cart</button>
                `;
                        itemsContainer.appendChild(card);
                    }
                }
            }
        }
        fetch(`./API/fetch_all_items_cate.php`).then(res => {
            return res.json();
        }).then(data => {
            displayItems(data);
        });

    </script>
